16. Tendo como dados de entrada a altura de uma pessoa, construa um 
algoritmo que calcule seu peso ideal, usando a seguinte fórmula: 

(72.7*altura) - 58

# include <stdio.h>

void main(){

    float h;

    printf("Entre com sua altura: \n");
    scanf("%f",&h);

    printf("Seu peso ideal eh: %f",((72.7*h) - 58));
}
