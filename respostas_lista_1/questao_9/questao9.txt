9. Faça um Programa que converta metros para centímetros.

# include <stdio.h>

int main(){

    float num;

    printf("Entre com o valor em metros a ser convertido para centimetros: \n");
    scanf("%f",&num);

    printf("O valor em centimetros eh: %f\n",(num*100));

    return 0;
}
